def to_s
  nombres_recursos = []
  for recurso in coleccion
    nombres_recursos << recurso.titulo
  end
  "- Id :#{@id_code} - nombre: #{@nombre} -categoria: #{@categoria} -coleccion:#{nombres_recursos.join(', ')},#{@precio}"
end
